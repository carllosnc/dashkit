import React, { useState } from 'react';
import { Pagination } from '../dashkit/Pagination';

# Pagination

A functional and responsive component for navigating between large sets of data or views.

## Quick Start
Provide the `currentPage`, `totalPages`, and an `onChange` callback to sync state perfectly.

<div className="not-prose border bg-card border-ds-200 dark:border-ds-800 ds-rounded flex flex-col gap-8 py-8 px-4 items-center justify-center w-full">
  {React.createElement(() => {
    const [page, setPage] = useState(1);
    return (
      <Pagination 
        currentPage={page} 
        totalPages={10} 
        onChange={setPage} 
      />
    );
  })}
</div>

```tsx
import { useState } from 'react';
import { Pagination } from '@/components/Pagination';

export function PaginationExample() {
  const [page, setPage] = useState(1);

  return (
    <Pagination 
      currentPage={page} 
      totalPages={10} 
      onChange={setPage} 
    />
  );
}
```

## Large Data Ranges
The `Pagination` component automatically calculates ellipsis (`...`) truncations for large ranges to prevent overflowing.

## Minimal Mode
By configuring `siblingCount={0}`, you can minimize the footprint of the pagination. The component defaults to `siblingCount={1}`.

<div className="not-prose border bg-card border-ds-200 dark:border-ds-800 ds-rounded flex flex-col gap-8 py-8 px-4 items-center justify-center w-full">
  {React.createElement(() => {
    const [page, setPage] = useState(25);
    return (
      <Pagination 
        currentPage={page} 
        totalPages={50} 
        siblingCount={0}
        onChange={setPage} 
      />
    );
  })}
</div>

```tsx
import { useState } from 'react';

export function PaginationMinimalExample() {
```
