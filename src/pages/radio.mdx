import { useState } from 'react';
import { Radio } from '../dashkit/Radio/Radio';

export const RadioDemo = () => {
  const [selected, setSelected] = useState('one');

  const options = [
    { id: 'one', label: 'Option One', description: 'This is the first selection.' },
    { id: 'two', label: 'Option Two', description: 'This is the second choice.' },
    { id: 'three', label: 'Option Three', description: 'Finally, the third option.' },
  ];

  return (
    <div className="flex flex-col gap-6 w-full max-w-sm mt-4">
      {options.map((opt) => (
        <Radio
          key={opt.id}
          label={opt.label}
          description={opt.description}
          name="demo-radio"
          value={opt.id}
          checked={selected === opt.id}
          onChange={() => setSelected(opt.id)}
        />
      ))}
    </div>
  );
}

# Radio

A premium radio button component for single choices from a set, with support for labels and descriptions.

## Import

```tsx
import { Radio } from '@/components/Radio/Radio'
```

## Installation

Add the radio button to your project using the Dashkit CLI:

```bash
bunx carllosnc/dashkit add radio
```

## Basic Usage

The basic radio button with a label. Usually part of a group sharing the same `name`.

<Preview>
  <div className="flex flex-col gap-4">
    <Radio label="Option 1" name="radio-demo" value="1" defaultChecked />
    <Radio label="Option 2" name="radio-demo" value="2" />
  </div>
</Preview>

```tsx
<div className="flex flex-col gap-4">
  <Radio label="Option 1" name="radio-group" value="1" defaultChecked />
  <Radio label="Option 2" name="radio-group" value="2" />
</div>
```

## Controlled

Works seamlessly with React's state management.

<Preview>
  <RadioDemo />
</Preview>

```tsx
import { useState } from 'react'

export function LanguageSelection() {
  const [value, setValue] = useState('react')

  return (
    <div className="flex flex-col gap-4">
      <Radio 
        label="React" 
        checked={value === 'react'} 
        onChange={() => setValue('react')} 
      />
      <Radio 
        label="Vue" 
        checked={value === 'vue'} 
        onChange={() => setValue('vue')} 
      />
    </div>
  )
}
```
