import { useState } from 'react';
import { DateField } from '../components/DateField/DateField';

export const BasicExample = () => {
  const [value, setValue] = useState(null);
  return (
    <div className="w-full max-w-sm">
      <DateField 
        label="Birth Date" 
        value={value} 
        onChange={setValue}
        description={value ? `Selected date: ${value.toLocaleDateString()}` : "Please enter your date of birth."}
      />
    </div>
  );
};

export const InteractiveDateField = () => {
  const [value, setValue] = useState(new Date());
  return (
    <div className="flex flex-col gap-4 w-full max-w-sm">
      <DateField 
        label="Event Date" 
        value={value} 
        onChange={setValue} 
        description={value ? `Selected: ${value.toLocaleDateString()}` : 'Please enter a date'}
      />
      <div className="flex gap-2">
        <button 
          onClick={() => setValue(new Date())}
          className="text-xs px-2 py-1 bg-ds-100 dark:bg-ds-800 rounded-md hover:bg-ds-200 dark:hover:bg-ds-700 transition-colors font-medium border dark:border-ds-700"
        >
          Reset to Today
        </button>
        <button 
          onClick={() => setValue(null)}
          className="text-xs px-2 py-1 bg-ds-100 dark:bg-ds-800 rounded-md hover:bg-ds-200 dark:hover:bg-ds-700 transition-colors font-medium border dark:border-ds-700"
        >
          Clear
        </button>
      </div>
    </div>
  );
};

# DateField

A premium, HeroUI-inspired input component for entering and editing date values with individual segments for Day, Month, and Year.

## Import

```tsx
import { DateField } from 'dashkit'
```

## Installation

Add the DateField component to your project using the Dashkit CLI:

```bash
bunx carllosnc/dashkit add datefield
```

## Basic Usage

A basic `DateField` with a label.

<Preview>
  <BasicExample />
</Preview>

```tsx
<DateField label="Birth Date" />
```

## With Description

Provide additional context using the `description` prop.

<Preview>
  <DateField 
    label="Event Date" 
    description="Select the date for your upcoming celebration." 
  />
</Preview>

```tsx
<DateField 
  label="Event Date" 
  description="Select the date for your upcoming celebration." 
/>
```

## States

`DateField` supports various states such as disabled and required.

<Preview>
  <div className="flex flex-col gap-6 w-full max-w-sm">
    <DateField label="Required Field" isRequired />
    <DateField label="Disabled Field" disabled value={new Date()} />
  </div>
</Preview>

```tsx
<DateField label="Required Field" isRequired />
<DateField label="Disabled Field" disabled value={new Date()} />
```

## Validation

Use the `error` prop to display validation feedback.

<Preview>
  <DateField 
    label="Expiry Date" 
    value={new Date('invalid')}
    error="Please enter a valid future date." 
  />
</Preview>

```tsx
<DateField 
  label="Expiry Date" 
  error="Please enter a valid future date." 
/>
```

## Controlled

Manage the date value explicitly using the `value` and `onChange` props.

<Preview>
  <InteractiveDateField />
</Preview>

```tsx
import { useState } from 'react'

export function ControlledDateField() {
  const [value, setValue] = useState(new Date())

  return (
    <DateField 
      label="Controlled Date" 
      value={value} 
      onChange={setValue} 
    />
  )
}
```
