import { useState } from 'react';
import { Chip } from '../components/dashkit/Chip/Chip';
import { FiClock, FiSettings } from 'react-icons/fi';

export const MultiSelectExample = () => {
  const [selectedChips, setSelectedChips] = useState(['React', 'Vite']);
  const toggleChip = (label) => {
    setSelectedChips(prev =>
      prev.includes(label) ? prev.filter(c => c !== label) : [...prev, label]
    );
  };
  return (
    <div className="flex flex-wrap gap-3">
      {['React', 'Next.js', 'Vite', 'Tailwind', 'Framer'].map(tech => (
        <Chip
          key={tech}
          label={tech}
          selected={selectedChips.includes(tech)}
          variant="outlined"
          onClick={() => toggleChip(tech)}
        />
      ))}
    </div>
  );
};

export const RemovableExample = () => {
  const [chips, setChips] = useState(['Personal', 'Work', 'Urgent', 'Low Priority']);
  const removeChip = (label) => {
    setChips(prev => prev.filter(c => c !== label));
  };
  return (
    <div className="flex flex-wrap gap-3">
      {chips.map(tag => (
        <Chip
          key={tag}
          label={tag}
          onDelete={() => removeChip(tag)}
        />
      ))}
      {chips.length === 0 && (
        <p className="text-xs text-ds-400 italic">No tags left.</p>
      )}
    </div>
  );
};

# Chip

A compact, interactive element that represents an input, attribute, or action, inspired by Material Design 3.

## Import

```tsx
import { Chip } from '@/components/Chip/Chip'
```

## Installation

Add the chip component to your project using the Dashkit CLI:

```bash
bunx carllosnc/dashkit add chip
```

## Basic Usage

The basic tonal chip represents a simple label or filter item.

<Preview>
  <div className="flex flex-wrap gap-4">
    <Chip label="Personal" />
    <Chip label="Work" />
    <Chip label="Urgent" />
  </div>
</Preview>

```tsx

<Chip label="Work" />
<Chip label="Urgent" />
```

## Multi-Selection

Selection state shows a checkmark icon automatically. This is useful for filter situations.

<Preview>
  <MultiSelectExample />
</Preview>

```tsx
import { useState } from 'react';

export const MultiSelectExample = () => {
  const [selected, setSelected] = useState(['React']);
  const toggle = (label) => {
    setSelected(prev =>
      prev.includes(label) ? prev.filter(c => c !== label) : [...prev, label]
    );
  };

  return (
    <div className="flex gap-3">
      {['React', 'Vite', 'Next.js'].map(tech => (
        <Chip
          key={tech}
          label={tech}
          selected={selected.includes(tech)}
          variant="outlined"
          onClick={() => toggle(tech)}
        />
      ))}
    </div>
  );
};
```

## Removable Labels

Chips can include a delete action, useful for tag systems.

<Preview>
  <RemovableExample />
</Preview>

```tsx
import { useState } from 'react';

export const RemovableExample = () => {
  const [chips, setChips] = useState(['Work', 'Personal']);
  const remove = (label) => setChips(prev => prev.filter(c => c !== label));

  return (
    <div className="flex gap-3">
      {chips.map(tag => (
        <Chip key={tag} label={tag} onDelete={() => remove(tag)} />
      ))}
    </div>
  );
};
```
