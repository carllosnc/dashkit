import { Input } from '../dashkit/Input/Input';
import { FiSearch, FiLock } from 'react-icons/fi';

export const PatternsExample = () => {
  const formatCPF = (val) => val
    .replace(/^(\d{3})(\d)/, '$1.$2')
    .replace(/^(\d{3})\.(\d{3})(\d)/, '$1.$2.$3')
    .replace(/\.(\d{3})(\d)/, '.$1-$2')
    .slice(0, 14);

  const formatPhone = (val) => {
    if (val.length <= 10) {
      return val.replace(/^(\d{2})(\d)/, '($1) $2').replace(/(\d{4})(\d)/, '$1-$2').slice(0, 14);
    }
    return val.replace(/^(\d{2})(\d)/, '($1) $2').replace(/(\d{5})(\d)/, '$1-$2').slice(0, 15);
  };

  return (
    <div className="flex flex-col gap-6 w-full max-w-sm">
      <Input label="CPF (Brasil)" placeholder="000.000.000-00" mask={/[^\d]/g} formatter={formatCPF} />
      <Input label="Phone Number" placeholder="(00) 00000-0000" mask={/[^\d]/g} formatter={formatPhone} />
    </div>
  );
};

# Input

A primitive input component with support for labels, icons, error states, and helper text.

## Import

```tsx
import { Input } from '@/components/Input/Input'
```

## Installation

Add the input component to your project using the Dashkit CLI:

```bash
bunx carllosnc/dashkit add input
```

## Basic Usage

The basic usage of the input with a label and placeholder.

<Preview>
  <Input label="Email Address" placeholder="Enter your email" className="max-w-sm" />
</Preview>

```tsx
<Input 
  label="Email Address" 
  placeholder="Enter your email" 
/>
```

## With Icons

Use `leftIcon` and `rightIcon` to add icons to the input. The icons automatically transition their color when the input is focused.

<Preview>
  <div className="flex flex-col gap-6 w-full max-w-sm">
    <Input label="Search" placeholder="Search documentation..." leftIcon={<FiSearch />} />
    <Input label="Password" type="password" placeholder="••••••••" rightIcon={<FiLock />} />
  </div>
</Preview>

```tsx
import { FiSearch, FiLock } from 'react-icons/fi'

<Input 
  label="Search" 
  placeholder="Search..." 
  leftIcon={<FiSearch />} 
/>

<Input 
  label="Password" 
  type="password" 
  rightIcon={<FiLock />} 
/>
```

## Masking & Formatting

The input component provides built-in support for masking (to filter characters) and formatting (to transform the displayed string).

### Digit Only Mask

<Preview>
  <div className="w-full max-w-sm">
    <Input label="Numeric Only" placeholder="Try typing letters..." mask={/[^\d]/g} />
  </div>
</Preview>

```tsx
<Input 
  label="Numeric Only" 
  mask={/[^\d]/g} 
  placeholder="123..." 
/>
```

### Complex Patterns

Using `mask` and `formatter` together enables masking for specific formats like Brasil CPF and phone numbers.

<Preview>
  <PatternsExample />
</Preview>

```tsx
const formatCPF = (val) => {
  return val
    .replace(/^(\d{3})(\d)/, '$1.$2')
    .replace(/^(\d{3})\.(\d{3})(\d)/, '$1.$2.$3')
    .replace(/\.(\d{3})(\d)/, '.$1-$2')
    .slice(0, 14);
};

<Input 
  label="CPF (Brasil)" 
  mask={/[^\d]/g}
  formatter={formatCPF}
  placeholder="000.000.000-00" 
/>
```
