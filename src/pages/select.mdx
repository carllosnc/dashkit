import { useState } from 'react';
import { Select } from '../components/dashkit/Select/Select';

export const SelectDemo = () => {
  const [value, setValue] = useState('');

  const shippingOptions = [
    { value: 'standard', label: 'Standard Shipping' },
    { value: 'priority', label: 'Priority Shipping' },
    { value: 'express', label: 'Express Delivery' },
    { value: 'overnight', label: 'Overnight Air' },
  ];

  return (
    <div className="flex flex-col gap-6 w-full max-w-sm mt-4 font-sans">
      <Select
        label="Shipping Method"
        description="Choose how you'd like your order delivered."
        placeholder="Choose method..."
        options={shippingOptions}
        value={value}
        onChange={setValue}
      />
      {value && (
        <div className="text-[11px] font-bold text-muted-foreground uppercase tracking-widest bg-ds-50 dark:bg-ds-800/50 p-3 rounded-lg border border-ds-200 dark:border-ds-800 animate-in fade-in duration-300">
          Selected Strategy: <span className="text-ds-900 dark:text-white font-black">{value}</span>
        </div>
      )}
    </div>
  );
}

# Select

A high-performance dropdown component for single choices, featuring portalled rendering to prevent layout overflow and full keyboard accessibility.

## Import

```tsx
import { Select } from '@/components/Select/Select'
```

## Installation

Add the select component using the Dashkit CLI:

```bash
bunx carllosnc/dashkit add select
```

## Examples

### Shipping Configuration

A standard implementation with labels, descriptions, and controlled state.

<Preview>
  <SelectDemo />

```tsx
export function ShippingForm() {
  const [method, setMethod] = useState('');
  
  return (
    <Select 
      label="Shipping Method"
      placeholder="Select method..."
      options={[
        { value: 'ground', label: 'Ground (3-5 days)' },
        { value: 'air', label: 'Air (Next day)' }
      ]}
      value={method}
      onChange={setMethod}
    />
  );
}
```

## Features

- **Portalled Rendering**: Renders the dropdown list using `createPortal` to the `document.body`, ensuring it stays on top of all other elements regardless of container overflow.
- **Dynamic Positioning**: Automatically calculates the trigger's position to align the popup accurately.
- **Interaction Guards**: Automatically closes the menu when the user scrolls or clicks outside to preserve UI intent.
- **Keyboard Friendly**: Fully supports `Enter`, `Space`, and `Escape` keys for navigation without a pointer.
- **Empty State Support**: Built-in messaging when no options are provided.
