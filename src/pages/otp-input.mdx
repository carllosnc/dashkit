import { useState } from 'react';
import { OtpInput } from '../components/dashkit/OtpInput/OtpInput';
import { Button } from '../components/dashkit/Button/Button';
import { FiCheckCircle } from 'react-icons/fi';

export const OtpInputDemo = () => {
  const [otp, setOtp] = useState('');
  const [isVerifying, setIsVerifying] = useState(false);
  const [isComplete, setIsComplete] = useState(false);

  const handleVerify = () => {
    setIsVerifying(true);
    setTimeout(() => {
      setIsVerifying(false);
      setIsComplete(true);
    }, 1500);
  };

  return (
    <div className="w-full max-w-lg p-8 rounded-2xl border border-ds-200 dark:border-ds-800 bg-card shadow-xl overflow-hidden mt-4">
      <div className="text-center mb-8">
        <h3 className="text-xl font-bold text-ds-900 dark:text-white mb-2">Two-Step Verification</h3>
        <p className="text-sm text-muted-foreground">
          Enter the 6-digit code sent to your device.
        </p>
      </div>

      <div className="flex flex-col items-center gap-8">
        <OtpInput 
          length={6} 
          value={otp} 
          onChange={setOtp} 
          disabled={isVerifying || isComplete}
        />

        {isComplete ? (
          <div className="flex items-center gap-2 text-emerald-500 font-semibold animate-in fade-in zoom-in duration-300">
            <FiCheckCircle className="size-5" />
            Verification successful
          </div>
        ) : (
          <Button 
            variant="filled" 
            onClick={handleVerify}
            disabled={otp.length !== 6 || isVerifying}
            loading={isVerifying}
          >
            Verify Code
          </Button>
        )}

        {!isComplete && (
          <button className="text-sm font-medium text-ds-400 hover:text-ds-900 dark:hover:text-white transition-colors">
            Didn't receive the code? <span className="underline decoration-dotted">Resend</span>
          </button>
        )}
      </div>
    </div>
  );
}

# OTP Input

A premium input component optimized for One-Time Passwords, featuring automatic focus shifting, digit masking, and full paste support.

## Import

```tsx
import { OtpInput } from '@/components/OtpInput/OtpInput';
```

## Installation

```bash
bunx carllosnc/dashkit add otp-input
```

## Features

- **Auto-Focus Logic**: Automatically moves focus to the next field as the user types.
- **Smart Backspace**: Deletes the current character and moves focus back when backspace is pressed.
- **Paste Support**: Intelligently handles pasted content by distributing digits across all input fields.
- **Input Filtering**: Only accepts numeric input to prevent accidental character entries.
- **Accessibility**: Includes proper `inputMode` and `autoComplete` attributes for mobile device optimization.
