  { value: 'nextjs', label: 'Next.js' },
  { value: 'sveltekit', label: 'SvelteKit' },
  { value: 'nuxt', label: 'Nuxt.js' },
  { value: 'remix', label: 'Remix' },
  { value: 'astro', label: 'Astro' },
  { value: 'gatsby', label: 'Gatsby' },
  { value: 'vite', label: 'Vite' },
  { value: 'webpack', label: 'Webpack' },
  { value: 'parcel', label: 'Parcel' },
  { value: 'rollup', label: 'Rollup' },
];

export const BasicExample = () => {
  const [value, setValue] = useState('nextjs');
  return (
    <div className="w-full max-w-md">
      <Combobox

# Combobox

A combobox is a flexible select component that allows users to filter a large list of options via a search input.

## Import

```tsx
import { Combobox } from '@/components/Combobox/Combobox'
```

## Installation

Add the combobox component to your project using the Dashkit CLI:

```bash
bunx carllosnc/dashkit add combobox
```

## Basic Search

The component behaves like a standard select, but with a built-in search input to handle long lists of data.

<Preview>
  <BasicExample />
</Preview>

```tsx
import { useState } from 'react';

const frameworks = [
  { value: 'nextjs', label: 'Next.js' },
  { value: 'vite', label: 'Vite' },
  { value: 'astro', label: 'Astro' },
];

export const BasicExample = () => {
  const [value, setValue] = useState('nextjs');
  return (
    <Combobox
      label="Frontend Framework"
      options={frameworks}
      value={value}
      onChange={setValue}
    />
  );
};
```

## Multi-Selection

Use the `multiple` prop to allow users to select more than one option. Selected items are displayed as removable chips.

<Preview>
  <MultiSelectExample />
</Preview>

```tsx
export const MultiSelectExample = () => {
  const [values, setValues] = useState(['nextjs', 'vite']);
  return (
    <Combobox
      label="Platform Tools"
      multiple
      options={frameworks}
      value={values}
      onChange={setValues}
    />
  );
};
```

You can add a `description` below the input to provide context or instructions.

```tsx
<Combobox
```

## States

The combobox supports different visual states like `disabled` or pre-filled values.

<Preview>
  <div className="flex flex-col gap-12 max-w-md w-full">
    <Combobox 
      label="Disabled State"
      disabled
      options={frameworks}
      value="nextjs"
    />
    <Combobox 
      label="Empty Search"
      placeholder="No options available..."
      options={[]}
    />
  </div>
</Preview>

```tsx
<Combobox label="Disabled" disabled value="nextjs" options={frameworks} />
<Combobox label="Empty" options={[]} />
```
