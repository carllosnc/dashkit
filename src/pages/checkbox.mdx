import { useState } from 'react';
import { Checkbox } from '../components/dashkit/Checkbox/Checkbox';

export const ControlledExample = () => {
  const [checked, setChecked] = useState(false);
  return (
    <div className="flex flex-col gap-4 w-full">
      <Checkbox 
        label="Controlled Checkbox" 
        description={`The current state is: ${checked ? 'Checked' : 'Unchecked'}`}
        checked={checked}
        onChange={(e) => setChecked(e.target.checked)}
      />
      <div className="flex gap-2">
        <button 
          onClick={() => setChecked(true)}
          className="text-xs px-2 py-1 bg-ds-100 dark:bg-ds-800 rounded-md hover:bg-ds-200 dark:hover:bg-ds-700 transition-colors"
        >
          Check
        </button>
        <button 
          onClick={() => setChecked(false)}
          className="text-xs px-2 py-1 bg-ds-100 dark:bg-ds-800 rounded-md hover:bg-ds-200 dark:hover:bg-ds-700 transition-colors"
        >
          Uncheck
        </button>
      </div>
    </div>
  );
};

# Checkbox

A custom-styled checkbox component for binary choices, with support for labels and descriptions.

## Import

```tsx
import { Checkbox } from '@/components/Checkbox/Checkbox'
```

## Installation

Add the checkbox component to your project using the Dashkit CLI:

```bash
bunx carllosnc/dashkit add checkbox
```

## Basic Usage

The basic checkbox with a label.

<Preview>
  <Checkbox label="Remember me" />
</Preview>

```tsx
<Checkbox label="Remember me" />
```

## Controlled

You can use the `checked` and `onChange` props to control the checkbox state with React.

<Preview>
  <ControlledExample />
</Preview>

```tsx
import { useState } from 'react'

export function ControlledCheckbox() {
  const [checked, setChecked] = useState(false)

  return (
    <Checkbox 
      label="Controlled Checkbox" 
      description={`Currently: ${checked ? 'Checked' : 'Unchecked'}`}
      checked={checked}
      onChange={(e) => setChecked(e.target.checked)}
    />
  )
}
```
