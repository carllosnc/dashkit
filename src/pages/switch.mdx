import { useState } from 'react';
import { Switch } from '../dashkit/Switch/Switch';

export const SwitchDemo = () => {
  const [enabled, setEnabled] = useState(false);

  return (
    <div className="flex flex-col gap-6 w-full max-w-sm mt-4">
      <Switch 
        label="Push Notifications" 
        description="Receive alerts for all incoming activity."
        checked={enabled}
        onChange={(e) => setEnabled(e.target.checked)}
      />
      <div className="text-sm text-muted-foreground bg-ds-50 dark:bg-white/5 p-4 rounded-xl border border-ds-200 dark:border-white/10">
        Status: <span className="font-bold text-foreground">{enabled ? 'Active' : 'Inactive'}</span>
      </div>
    </div>
  );
}

# Switch

A high-performance, premium toggle switch component for binary choices, with support for labels and descriptions.

## Import

```tsx
import { Switch } from '@/components/Switch/Switch'
```

## Installation

Add the switch component to your project using the Dashkit CLI:

```bash
bunx carllosnc/dashkit add switch
```

## Basic Usage

The basic switch with a label.

<Preview>
  <Switch label="Enable Tracking" />
</Preview>

```tsx
<Switch label="Enable Tracking" />
```

## With Description

You can add descriptions to provide more context for the toggle activity.

```tsx
<Switch 
  label="Background Updates"

/>
```

## Controlled State

Manage the switch state using React's `useState`.

<Preview>
  <SwitchDemo />
</Preview>

```tsx
import { useState } from 'react'

export function Settings() {
  const [enabled, setEnabled] = useState(false)

  return (
    <Switch 
      label="Push Notifications" 
      checked={enabled} 
      onChange={(e) => setEnabled(e.target.checked)} 
    />
  )
}
```

## States

Supports checked, disabled, and focus states.

<Preview>

  </div>

```tsx
<Switch label="Default Checked" defaultChecked />
```
